ALTER TABLE ARTWORK_CLONE DROP COLUMN MEDIUM_DISPLAY;
ALTER TABLE ARTWORK_CLONE ADD COLUMN PARITY VARCHAR(10);

DELETE FROM ARTWORK_CLONE
WHERE ID = 'id';

UPDATE ARTWORK_CLONE
SET PARITY = CASE
    WHEN CAST(id AS INTEGER) % 2 = 0 THEN 'even'
    ELSE 'odd'
END;

UPDATE ARTWORK_CLONE
SET TITLE = SPLIT_PART(title, ',', 1);

UPDATE ARTWORK_CLONE
SET TITLE = REGEXP_REPLACE(SPLIT_PART(title, ',', 1), '\\s*\\(.*?\\)', '');

UPDATE artwork_clone
SET artist = REGEXP_REPLACE(artist, '[^A-Za-z].*$', '');

DELETE FROM ARTWORK_CLONE
WHERE ID IS NULL
   OR TITLE IS NULL
   OR ARTIST IS NULL
   OR DATE IS NULL
   OR IMAGE_URL IS NULL
   OR PARITY IS NULL;

SELECT * FROM ARTWORK_CLONE;
